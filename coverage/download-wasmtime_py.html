1# Helper script to download a precompiled binary of the wasmtime dll for the 

2# current platform. Currently always downloads the dev release of wasmtime. 

3 

4import urllib.request 

5import zipfile 

6import tarfile 

7import io 

8import sys 

9 

10 

11def main(): 

12 is_zip = False 

13 if sys.platform == 'linux': 

14 filename = 'wasmtime-dev-x86_64-linux-c-api.tar.xz' 

15 elif sys.platform == 'win32': 

16 filename = 'wasmtime-dev-x86_64-windows-c-api.zip' 

17 is_zip = True 

18 elif sys.platform == 'darwin': 

19 filename = 'wasmtime-dev-x86_64-macos-c-api.tar.xz' 

20 else: 

21 raise RuntimeError("unknown platform: " + sys.platform) 

22 

23 url = 'https://github.com/bytecodealliance/wasmtime/releases/download/dev/' 

24 url += filename 

25 print('Download', url) 

26 

27 with urllib.request.urlopen(url) as f: 

28 contents = f.read() 

29 

30 if is_zip: 

31 t = zipfile.ZipFile(io.BytesIO(contents)) 

32 for member in t.namelist(): 

33 if not member.endswith('.dll'): 

34 continue 

35 contents = t.read(member) 

36 with open("wasmtime/wasmtime.pyd", "wb") as f: 

37 f.write(contents) 

38 sys.exit(0) 

39 else: 

40 t = tarfile.open(fileobj=io.BytesIO(contents)) 

41 for member in t.getmembers(): 

42 linux_so = member.name.endswith('.so') 

43 macos_dylib = member.name.endswith('.dylib') 

44 if not linux_so and not macos_dylib: 

45 continue 

46 contents = t.extractfile(member).read() 

47 with open("wasmtime/wasmtime.pyd", "wb") as f: 

48 f.write(contents) 

49 sys.exit(0) 

50 

51 raise RuntimeError("failed to find dynamic library")

54if __name__ == '__main__':
